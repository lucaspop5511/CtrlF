struct Node {
    int data; 
    struct Node* next; 
}; 
void insert(struct Node** head, int newData) { 
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node)); 
    newNode->data = newData; 
    newNode->next = (*head); 
    (*head) = newNode; 
}
